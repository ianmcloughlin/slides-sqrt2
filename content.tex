\begin{frame}

  \[\sqrt{2} = \frac{a}{b} \] \pause
  
  \[gcd(a,b) = 1\] \pause
  
  \[2 = \frac{a^2}{b^2} \] \pause

  \[2b^2 = a^2 \] \pause

  \[2 \mid a^2 \Rightarrow 2 \mid a\] \pause

\end{frame}
